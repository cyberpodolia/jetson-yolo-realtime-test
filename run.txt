## 1) Export ONNX on the training machine
# fair 320 export for Jetson A/B against the old 320 baseline
.venv\Scripts\yolo.exe export model=runs/detect/train4/weights/best.pt format=onnx imgsz=320 simplify=True
Copy-Item runs\detect\train4\weights\best.onnx artifacts\train4_320.onnx

# optional separate 640 export if you want a higher-res non-baseline deployment build
.venv\Scripts\yolo.exe export model=runs/detect/train4/weights/best.pt format=onnx imgsz=640 simplify=True
Copy-Item runs\detect\train4\weights\best.onnx artifacts\train4_640.onnx

## 2) Sync refreshed repo/artifacts to Jetson
# git pull on Jetson, or copy refreshed ONNX artifacts into the repo there
# minimum for current fair A/B path: `artifacts/train4_320.onnx`
# optional baseline rebuild input: `artifacts/joystick.onnx`

## 3) Performance mode on Jetson host
sudo nvpmodel -m 0
sudo jetson_clocks

## 4) Verify ONNX exists on Jetson
ls -lh /home/kolins/jetson-yolo/artifacts/train4_320.onnx

## 5) Build TensorRT engine
cd /home/kolins/jetson-yolo
bash scripts/build_engine.sh artifacts/train4_320.onnx artifacts/train4_320_fp16.engine
bash scripts/build_engine.sh artifacts/joystick.onnx artifacts/joystick_320_fp16_native.engine

## 6) Quick TensorRT sanity (engine-only benchmark)
/usr/src/tensorrt/bin/trtexec --loadEngine=artifacts/train4_320_fp16.engine --warmUp=1000 --duration=10 --streams=1

## 7) Runtime via Docker (recommended)
sudo docker run --rm -it \
  --runtime nvidia --network host --ipc host \
  --device /dev/video0:/dev/video0 \
  -v /home/kolins/jetson-yolo:/workspace \
  -w /workspace \
  nvcr.io/nvidia/l4t-ml:r32.7.1-py3 \
  python3 -m src.app \
    --engine artifacts/train4_320_fp16.engine \
    --camera 0 --frames 300 --imgsz 320 \
    --conf 0.6 --iou 0.7 --max-det 10 \
    --tracker csrt --det-interval 3 --track-interval 2 \
    --save-json outputs/metrics_det3.json

## 8) Benchmark matrix det_interval={1,3,5} in Docker
for N in 1 3 5; do

      --conf 0.6 --iou 0.7 --max-det 10 \
